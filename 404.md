<div class="alert alert-warning" role="alert" id="not-found">
<p>
Tässä osoitteessa ei ole sivua. Jos kirjoitit osoitteen käsin
tai kopioit esim. sähköpostista, voit tarkistaa sen oikeinkirjoituksen.
Muuten voit etsiä sivua <a href="/" class="alert-link">pääsivulta</a> alkaen.
</p>
</div>
<div class="alert alert-info d-none" role="alert" id="redirect-notice">
<p>
Tämä tiedosto on siirretty Google Driveen. Sinut ohjataan sinne hetken kuluttua.
Jos uudelleenohjaus ei toimi, <a id="redirect-link" href="#" class="alert-link">klikkaa tästä</a>.
</p>
</div>
<script>
(function() {
  var p = decodeURIComponent(window.location.pathname);
  if (!p.endsWith(".pdf")) return;
  fetch("/pdf-redirects.json")
    .then(function(r) { return r.json(); })
    .then(function(redirects) {
      var id = redirects[p];
      if (!id) return;
      var url = "https://drive.google.com/file/d/" + id + "/view";
      document.getElementById("not-found").classList.add("d-none");
      document.getElementById("redirect-notice").classList.remove("d-none");
      document.getElementById("redirect-link").href = url;
      setTimeout(function() { window.location.replace(url); }, 1500);
    });
})();
</script>
